from mrjob.job import MRJob
import re

WORD_REGEXP = re.compile(r"[a-zA-Z]+")

class MRWordCount(MRJob):

    def mapper(self, _, line):
        for word in WORD_REGEXP.findall(line):
            yield word.lower(), 1

    def reducer(self, word, counts):
        yield word, sum(counts)

if __name__ == "__main__":
    MRWordCount.run()

from collections import defaultdict

def pagerank_mapreduce(graph, damping=0.85, max_iterations=100, tolerance=1e-6):
    num_nodes = len(graph)
    
    
    ranks = {node: 1 / num_nodes for node in graph}
    
    for iteration in range(max_iterations):
        
      
        contributions = []
        
        for node in graph:
            if len(graph[node]) == 0:
                # Dangling node distributes rank to all nodes
                for target in graph:
                    contributions.append((target, ranks[node] / num_nodes))
            else:
                share = ranks[node] / len(graph[node])
                for target in graph[node]:
                    contributions.append((target, share))
        
        
        grouped = defaultdict(list)
        for node, value in contributions:
            grouped[node].append(value)
        
        
        new_ranks = {}
        for node in graph:
            rank_sum = sum(grouped[node])
            new_ranks[node] = (1 - damping) / num_nodes + damping * rank_sum
        
        
        diff = sum(abs(new_ranks[node] - ranks[node]) for node in graph)
        ranks = new_ranks
        
        if diff < tolerance:
            print(f"Converged after {iteration + 1} iterations.")
            break
    
    return ranks

graph = {
    'A': ['B', 'C'],
    'B': ['C'],
    'C': ['A'],
    'D': ['C']
}

ranks = pagerank_mapreduce(graph)

print("\nFinal PageRank Scores:")
for node, rank in ranks.items():
    print(f"{node}: {rank:.4f}")

import numpy as np
from collections import defaultdict

# Generate Sample Dataset

np.random.seed(42)
data = np.vstack((
    np.random.randn(100, 2) + [2, 2],
    np.random.randn(100, 2) + [8, 8]))

K = 2
max_iters = 10
epsilon = 0.001

# Initialize Centroids

centroids = data[np.random.choice(len(data), K, replace=False)]

# Distance Function

def euclidean_distance(a, b):
    return np.linalg.norm(a - b)

# MAP FUNCTION

def mapper(data_chunk, centroids):
    mapped = []
    for point in data_chunk:
        distances = [euclidean_distance(point, c) for c in centroids]
        nearest_centroid = np.argmin(distances)
        mapped.append((nearest_centroid, point))
    return mapped

# REDUCE FUNCTION

def reducer(mapped_data):
    cluster_dict = defaultdict(list)

    # Shuffle + Sort (group by key)
    for key, point in mapped_data:
        cluster_dict[key].append(point)

    new_centroids = []
    for key in sorted(cluster_dict.keys()):
        new_centroids.append(np.mean(cluster_dict[key], axis=0))

    return np.array(new_centroids)

# Iterative MapReduce KMeans

for iteration in range(max_iters):
    print(f"Iteration {iteration+1}")

    # MAP
    mapped = mapper(data, centroids)

    # REDUCE
    new_centroids = reducer(mapped)

    # Check convergence
    shift = np.linalg.norm(new_centroids - centroids)
    print("Centroid shift:", shift)

    if shift < epsilon:
        break

    centroids = new_centroids

print("\nFinal Centroids:")
print(centroids)
